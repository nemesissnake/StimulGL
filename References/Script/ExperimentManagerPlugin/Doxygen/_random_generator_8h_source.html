00037 // This template class combines two different random number generators
00038 // for improved randomness. R1 and R2 are any two different random number
00039 // generator classes.
00040 template <class RG1, class RG2>
00041 class TRandomCombined : private RG1, private RG2 
00042 {
00043 public:
00044         TRandomCombined(int seed) : RG1(seed), RG2(seed+1) {};
00045 
00046         void RandomInit(int seed) {        // re-seed
00047                 RG1::RandomInit(seed);
00048                 RG2::RandomInit(seed+1);
00049         }
00050 
00051         double Random() {
00052                 double r = RG1::Random() + RG2::Random();
00053                 if (r >= 1.) r -= 1.;
00054                 return r;
00055         }
00056 
00057         int IRandom(int min, int max)
00058         {// output random integer
00059                 if(min == max)
00060                         return min;
00061                 // get integer random number in desired interval
00062                 int iinterval = max - min;// + 1;
00063                 if (iinterval < 0)//<= 0) 
00064                         return 0x80000000; // error
00065                 int r = int((iinterval+1) * Random()); // truncate, Random() returns <1.0
00066                 //if (r >= iinterval) 
00067                 //      return max;//r = iinterval-1;
00068                 return min + r;
00069         }
00070 
00071         double DRandom(double min, double max)
00072         {// output random double
00073                 if(min == max)
00074                         return min;
00075                 // get double random number in desired interval
00076                 double iinterval = max - min;// + 1;
00077                 if (iinterval <= 0)//<= 0) 
00078                         return 0x80000000; // error
00079                 double dsmall = DBL_MIN;
00080                 double r = (iinterval+DBL_MIN) * Random(); // truncate
00081                 //if (r >= iinterval) 
00082                 //      return max;//   r = iinterval-1;
00083                 return min + r;
00084         }
00085 };
